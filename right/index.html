Simple Sidebar

Demo Right

A simple jQuery sidebar plugin.

Download

git clone https://github.com/dcdeiv/simple-sidebar
npm install simple-sidebar
bower install simple-sidebar

Example of Right Options

	        
$( document ).ready(function() {
    
    $.fn.simpleSidebar.settings.opener = "#toggle-sidebar";
    $.fn.simpleSidebar.settings.wrapper = "#main";
    $.fn.simpleSidebar.settings.animation.easing = "easeOutQuint";
    $.fn.simpleSidebar.settings.sidebar.align = "right";
    $.fn.simpleSidebar.settings.sidebar.closingLinks = ".close-sb";
    $.fn.simpleSidebar.settings.sidebar.subWrapper = false;
        
    $( '#sidebar' ).simpleSidebar();
});
